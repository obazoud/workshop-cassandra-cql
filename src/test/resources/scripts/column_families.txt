/* Column Family Users */

create table user(
    id uuid PRIMARY KEY
    , name varchar
    , email varchar
    , password varchar
    , age int
);

/* Column Family Tracks */

create table tracks (
    id timeuuid PRIMARY KEY
    , title varchar
    , release date
    , duration float
    , tags set<varchar>
);

/* Column Family Click Stream */

create table user_click_stream (
    user_id uuid
    , when timestamp
    , url varchar
    , referer varchar
    , PRIMARY KEY(user_id, when, url)
);

/* Column Family Liked Tracks */

create table track_likes (
    user_id uuid PRIMARY KEY
    , track_id timeuuid
) USE COMPACT STORAGE;
